const express = require('express');
const router  = express.Router();

let users = [
  { id: 1, name: 'يوسف', email: 'yousef@example.com' },
];

// GET  — جلب كل المستخدمين
router.get('/', (req, res) => {
  res.json({ success: true, data: users });
});

// POST — إضافة مستخدم جديد
router.post('/', (req, res) => {
  const user = { id: Date.now(), ...req.body };
  users.push(user);
  res.status(201).json({ success: true, data: user });
});

// PUT — تعديل مستخدم
router.put('/:id', (req, res) => {
  const idx = users.findIndex(u => u.id == req.params.id);
  if (idx === -1) return res.status(404).json({ error: 'غير موجود' });
  users[idx] = { ...users[idx], ...req.body };
  res.json({ success: true, data: users[idx] });
});

// DELETE — حذف مستخدم
router.delete('/:id', (req, res) => {
  users = users.filter(u => u.id != req.params.id);
  res.json({ success: true, message: 'تم الحذف' });
});

// Middleware للتحقق من الـ Token
const protect = (req, res, next) => {
  const auth = req.headers.authorization;
  if (!auth?.startsWith('Bearer '))
    return res.status(401).json({ error: 'غير مصرح' });
  try {
    req.user = jwt.verify(auth.split(' ')[1], SECRET);
    next();
  } catch { res.status(401).json({ error: 'Token غير صالح' }); }
};

// Route محمي
router.get('/profile', protect, (req, res) => {
  res.json({ userId: req.user.id });
});

const express  = require('express');
const morgan   = require('morgan');
const helmet   = require('helmet');
const cors     = require('cors');
const rateLimit = require('express-rate-limit');

const app = express();

// ── أمان HTTP Headers
app.use(helmet());

// ── السماح بـ Cross-Origin Requests
app.use(cors({ origin: 'https://myapp.com' }));

// ── تحديد عدد الطلبات (Rate Limiting)
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 دقيقة
  max: 100,
  message: 'طلبات كثيرة جداً، حاول لاحقاً'
});
app.use('/api', limiter);

// ── تسجيل الطلبات في التيرمينال
app.use(morgan('dev'));

// ── قراءة JSON في الـ Body
app.use(express.json({ limit: '10kb' }));

// ── Middleware مخصص: قياس الوقت
app.use((req, res, next) => {
  req.startTime = Date.now();
  res.on('finish', () => {
    const ms = Date.now() - req.startTime;
    console.log(`[${req.method}] ${req.url} — ${ms}ms`);
  });
  next();
});

// ── معالجة الأخطاء العامة
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ error: 'حدث خطأ في الخادم' });
});

app.listen(3000);

دورة حياة الطلب

كل طلب HTTP يمر بسلسلة من Middleware قبل أن تصل الاستجابة للمستخدم

1
استقبال الطلب
Express يستقبل الـ HTTP Request ويحوله لكائن req غني بالمعلومات
app.use((req, res, next) => { ... })
2
Middleware الأمان
helmet, cors, rate-limiter تفحص وتحمي كل طلب قبل المعالجة
app.use(helmet(), cors(), limiter)
3
تحليل الـ Body
express.json() يحول نص JSON في body الطلب إلى JavaScript object
app.use(express.json())
4
التوثيق والصلاحيات
التحقق من الـ JWT Token أو Session قبل الوصول للمصادر الحساسة
const protect = (req, res, next) => { ... }
5
Route Handler
يصل الطلب أخيراً للـ Handler الصحيح الذي يعالج المنطق ويبني الاستجابة
app.get('/users', async (req, res) => { ... })
✓
إرسال الاستجابة
res.json() أو res.render() أو res.send() يُرسلون البيانات للـ Client
res.status(200).json({ success: true })
